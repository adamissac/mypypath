from datetime import datetime

def greet(name: str) -> str:
    return f"Hello, {name}!"

def times(n: int) -> list[int]:
    return [i * 2 for i in range(n)]

if __name__ == "__main__":
    print(greet("World"))
    print("Even numbers:", times(5))
    print("Run at:", datetime.now().strftime("%H:%M:%S"))
